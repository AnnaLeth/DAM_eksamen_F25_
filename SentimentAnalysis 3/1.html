library(tidytext)
library(ggwordcloud)
library(readtext)
danske_taler<-tibble(readtext("FP_work/data_taler/*pdf"))
danske_taler_stninger <- danske_taler %>% 
  unnest_sentences(sentence, text)

danske_taler_ord <- danske_taler %>% 
  unnest_tokens(word, text)
nationalistiske_ord <- c("danmark", "dansker", "dansk",
                         "grnser", "grnse", "grnsekontrol", "samfund", "tradition", "os", "vores", "dem", "deres", "regeringen", "velfrdssamfundet", "bevare" )
danske_taler_total_ord <- danske_taler_ord %>% 
  group_by(doc_id) %>% 
  count(word, sort = TRUE) %>% 
  summarise(total_ord = sum(n))
danske_taler_nat_ord_vis<- danske_taler_ord %>% 
  group_by(doc_id) %>% 
  count(word, sort = TRUE) %>% 
  left_join(danske_taler_total_ord, by = "doc_id") %>%
  ungroup() %>% 
  mutate(procentdel = n/total_ord) %>% 
  filter(word %in%nationalistiske_ord)%>% 
  ggplot(aes(x = reorder(doc_id, procentdel), y = procentdel, fill = doc_id)) +
  geom_col() +
  facet_wrap(~word, ncol = 3) +
  coord_flip() +  # Gr det nemmere at lse ord
  labs(
    title = "Nationalistiske ords fremkomst i procent",
    x = "procentvis forekomst i given tale",
    y = "nationalistiske ord"
  ) +
  theme_minimal()

print(danske_taler_nat_ord_vis)

ggsave("visualisering/danske_taler_nat_ord_vis.png", danske_taler_nat_ord_vis, bg = "white")
## Saving 7 x 5 in image

#Opsummering: Analyse 1 - Vi har lavet en illustration, der viser forekomsten af de forskellige nationalistiske ord, vi har defineret, i procentdel, da talerne er af meget forskellig lngde. Visualiserings styrke er, at den kan visualisere, hvilke ord der nvnes blandt partiformnd i deres taler. Visualiseringen viser ogs, hvor hyppigt hvert af de nationalistiske ord nvnes af de respektive partier i deres taler.

library(Sentida)

Sys.setlocale("LC_ALL", "da_DK.UTF-8")

danske_taler_sentida <- danske_taler_stninger %>% 
  rowwise() %>% 
  mutate(sentida_score = sentida(sentence)) 
danske_taler_sentida %>% 
  group_by(doc_id) %>% 
  summarise(gennemsnit_sentida = mean(sentida_score))

str_c(nationalistiske_ord,collapse = "|")
## [1] "danmark|dansker|dansk|grnser|grnse|grnsekontrol|samfund|tradition|os|vores|dem|deres|regeringen|velfrdssamfundet|bevare"
total_nat_sentida <- danske_taler_stninger %>% 
  filter(str_detect(sentence,"danmark|dansker|dansk|grnser|grnse|grnsekontrol|samfund|tradition|os|vores|dem|deres|regeringen|velfrdssamfundet|bevare")) %>% 
  rowwise() %>% 
  mutate(sentida_score = sentida(sentence)) %>% 
  group_by(doc_id) %>% 
  summarise(total_nat_sentida = mean(sentida_score)) %>% 
  arrange(desc(total_nat_sentida))
ggplot(total_nat_sentida, aes(x = doc_id, y = total_nat_sentida, fill = doc_id)) + 
  geom_bar(stat = "identity") +
  theme_minimal() + 
  labs(
    title = "Talernes gennemsnitlige sentimentscore for nationalistiske ord", 
    x = "parti",
    y = "Gennemsnitlig sentiment score"
  )

ggsave("visualisering/total_nat_sentida.png")
